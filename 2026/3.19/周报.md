## 部署大模型

部署了qwen2.5:7b模型，qwen3的模型好像还没有开源，下载不了。

目前用json格式的测试文本测试了一下，可以基于几个关键词生成完整的一段描述文字。

```python
import requests
import json
url = "http://localhost:11434/api/generate"
def test(features):
    prompt = f"""
    根据以下关键词生成一句完整的英文图像描述。
    只返回JSON格式：
    {{
      "description": "..."
    }}

    关键词: {', '.join(features)}
    """
    payload = {
        "model": "qwen2.5:7b",
        "prompt": prompt,
        "stream": False
    }
    response = requests.post(url, json=payload)
    data = response.json()
    content = data["response"]
    # 把字符串解析成真正JSON
    result = json.loads(content)
    print("最终JSON:", result)
    print("描述:", result["description"])
    print("-"*60)
test(["sunset", "beach", "two people", "walking", "silhouette"])
test(["dog", "brown", "grass", "sunny", "running"])
test(["cat", "window", "rain", "sleeping"])
```
